\documentclass[tikz,border=10pt]{standalone}
\usepackage{tikz}
\usepackage{pifont}
\usetikzlibrary{shapes,arrows.meta,positioning,calc,fit,backgrounds}

\begin{document}
\begin{tikzpicture}[
    node distance=0.85cm,
    % --- Color palette ---
    inputColor/.style={fill=blue!8, draw=blue!40, thick},
    tcnColor/.style={fill=teal!10, draw=teal!50, thick},
    auaaColor/.style={fill=orange!15, draw=orange!70, thick},
    adapterColor/.style={fill=purple!12, draw=purple!65, thick},
    attribColor/.style={fill=green!12, draw=green!60, thick},
    outputColor/.style={fill=green!18, draw=green!70, very thick},
    % --- Block styles ---
    block/.style={rectangle, rounded corners=3pt, draw, minimum width=2.6cm, minimum height=0.9cm, align=center, font=\footnotesize\bfseries},
    smallBlock/.style={rectangle, rounded corners=2pt, draw, minimum width=2.0cm, minimum height=0.7cm, align=center, font=\footnotesize},
    tinyBlock/.style={rectangle, rounded corners=2pt, draw, minimum width=1.5cm, minimum height=0.55cm, align=center, font=\scriptsize},
    arrow/.style={-{Stealth[scale=0.8]}, thick, draw=gray!50},
    feedbackArrow/.style={-{Stealth[scale=0.7]}, thick, draw=red!60, densely dashed},
    % Callout: transparent fill so it does NOT cover the inner block
    callout/.style={rectangle, rounded corners=5pt, fill=none, draw=red!50, thick, dashed, inner sep=4pt, font=\footnotesize},
    highlight/.style={font=\footnotesize\bfseries, text=red!70!black},
]

% ============================================================
% INPUT LAYER
% ============================================================
\node [block, inputColor, minimum width=3.2cm, minimum height=1.3cm] (input) {
    \textbf{Multivariate Sensor Input}\\[2pt]
    \scriptsize $\mathbf{X} \in \mathbb{R}^{W \times n}$ \\[1pt]
    \scriptsize CO$_2$ | VOC | Temp | Hum | PM$_{2.5}$
};

% ============================================================
% FEATURE ENGINEERING
% ============================================================
\node [smallBlock, inputColor, below=of input] (features) {
    \textbf{Windowed Features}\\[1pt]
    \scriptsize $x_f(t), \Delta_f, \mu_f^W, \sigma_f^W$
};

% ============================================================
% TCN BACKBONE (3 layers)
% ============================================================
\node [block, tcnColor, below=of features, minimum width=6.0cm, minimum height=2.0cm] (tcn) {
    \textbf{TCN Backbone (3 Layers)}\\[4pt]
    \begin{tikzpicture}[scale=0.7, every node/.style={font=\tiny}]
        \node [tinyBlock, fill=white, draw=teal!40] (l1) {Layer 1\\d=1, ch=64};
        \node [tinyBlock, fill=white, draw=teal!40, right=0.3cm of l1] (l2) {Layer 2\\d=2, ch=64};
        \node [tinyBlock, fill=white, draw=teal!40, right=0.3cm of l2] (l3) {Layer 3\\d=4, ch=64};
        \draw [-{Stealth[scale=0.5]}, teal!50] (l1) -- (l2);
        \draw [-{Stealth[scale=0.5]}, teal!50] (l2) -- (l3);
    \end{tikzpicture}\\[2pt]
    {\scriptsize Causal Conv + WeightNorm + ReLU + Stochastic Depth}
};

% ============================================================
% MC DROPOUT (stochastic passes)
% ============================================================
\node [smallBlock, tcnColor, below=of tcn, minimum width=4.5cm] (mcdropout) {
    \textbf{MC Dropout ($M$=50 passes)}\\[1pt]
    \scriptsize $\{\hat{\mathbf{y}}^{(1)},...,\hat{\mathbf{y}}^{(M)}\} \to \hat{\mathbf{y}}, \boldsymbol{\sigma}$
};

% ============================================================
% INNOVATION 1: AUAA (below-left of MC dropout)
% ============================================================
\node [block, auaaColor, below=2.0cm of mcdropout, xshift=-4.0cm, minimum width=5.2cm, minimum height=1.7cm] (auaa) {
    \textbf{Innovation 1: AUAA}\\[3pt]
    \scriptsize $\boldsymbol{\alpha}_t = \text{softmax}\!\left(\frac{\mathbf{Q}\mathbf{K}^\top}{\sqrt{d_k}}\right) \odot \frac{1}{1+\boldsymbol{\sigma}_t}$\\[3pt]
    {\scriptsize Uncertainty gates attention weights}
};

% ============================================================
% INNOVATION 2: Dynamic Weight Adapter (below-right of MC dropout)
% ============================================================
\node [block, adapterColor, below=2.0cm of mcdropout, xshift=4.0cm, minimum width=5.2cm, minimum height=1.7cm] (adapter) {
    \textbf{Innovation 2: Dynamic Weight Adapter}\\[3pt]
    \scriptsize $s_t = \alpha_t \cdot \mathbf{e}_t + (1-\alpha_t) \cdot \mathbf{u}_t$\\[3pt]
    {\scriptsize GRU-context learns $\alpha_t$ over time}
};

% ============================================================
% INNOVATION 3: Feature Attribution (below MC dropout, centred)
% ============================================================
\node [block, attribColor, below=2.0cm of mcdropout, yshift=-1.8cm, minimum width=5.2cm, minimum height=1.7cm] (attribution) {
    \textbf{Innovation 3: Feature Attribution}\\[3pt]
    \scriptsize $\mathbf{w}_f = \text{softmax}(\mathbf{V}\!\cdot\!\tanh(\mathbf{W}\mathbf{x}+\mathbf{b}))$\\[3pt]
    {\scriptsize Per-sensor importance + Chebyshev bound}
};

% ============================================================
% OUTPUT (below attribution)
% ============================================================
\node [block, outputColor, below=0.8cm of attribution, minimum width=8.0cm, minimum height=1.4cm] (output) {
    \textbf{Output}\\[3pt]
    {\scriptsize Anomaly Score $s_t$ \quad | \quad Per-sensor attribution $\mathbf{w}_f$}\\[3pt]
    {\scriptsize $P(\text{FP}) \leq 1/k^2$ \,(Chebyshev distribution-free FP guarantee)}
};

% ============================================================
% ARROWS (main flow)
% ============================================================
\draw [arrow] (input) -- (features);
\draw [arrow] (features) -- (tcn);
\draw [arrow] (tcn) -- (mcdropout);

% MC dropout splits into AUAA + Adapter
\draw [arrow] (mcdropout.south) -- ++(0,-0.3) -| (auaa.north);
\draw [arrow] (mcdropout.south) -- ++(0,-0.3) -| (adapter.north);

% Both innovations feed into attribution
\draw [arrow] (auaa.south) |- (attribution.west);
\draw [arrow] (adapter.south) |- (attribution.east);

% Attribution to output
\draw [arrow] (attribution.south) -- (output.north);

% ============================================================
% FEEDBACK LOOPS (uncertainty-aware integration)
% ============================================================
% Uncertainty feedback loop: AUAA back to TCN attention
\draw [feedbackArrow]
    (auaa.west) -- ++(-1.2,0)
    node [pos=0.5, above, font=\tiny\bfseries, text=red!70] {uncertainty $\to$ attention}
    |- (tcn.west);

% GRU memory loop in adapter
\draw [feedbackArrow]
    (adapter.east) -- ++(1.2,0)
    node [pos=0.5, right, font=\tiny\bfseries, text=red!70] {GRU memory}
    -- ++(0,1.5)
    -- ($(adapter.north east)+(1.2,0.4)$)
    -- (adapter.north east);

% ============================================================
% INNOVATION CALLOUT LABELS (transparent, on background layer)
% ============================================================
\begin{pgfonlayer}{background}
    \node [callout, fit=(auaa)] {};
    \node [callout, fit=(adapter)] {};
    \node [callout, fit=(attribution)] {};
\end{pgfonlayer}

% Gap labels next to each innovation
\node [highlight, above=4pt of auaa.north west, anchor=south west] {Gap 1: Sensor reliability};
\node [highlight, above=4pt of adapter.north west, anchor=south west] {Gap 2: Context-aware scoring};
\node [highlight, below=4pt of attribution.south west, anchor=north west] {Gap 3: Actionable per-sensor output};

% ============================================================
% MINI ABLATION TABLE (fresh data from research_outputs_full)
% ============================================================

       right=1.0cm of output, minimum width=4.6cm, inner sep=6pt, font=\scriptsize] (table) {

};

% ============================================================
% KEY INSIGHT TEXT
% ============================================================
\node [font=\small\itshape, text=red!50!black, below=0.4cm of output] (insight) {
    \textbf{Key insight:} uncertainty is not just a score --- it reshapes attention and gates scoring.
};
