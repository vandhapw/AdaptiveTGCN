% ============================================================
% MC DROPOUT (stochastic passes)
% ============================================================
\node [smallBlock, tcnColor, below=of tcn, minimum width=4.5cm] (mcdropout) {
    \textbf{MC Dropout ($M$=50 passes)}\\[1pt]
    \scriptsize $\{\hat{\mathbf{y}}^{(1)},...,\hat{\mathbf{y}}^{(M)}\} \to \hat{\mathbf{y}}, \boldsymbol{\sigma}$
};

% ============================================================
% AUAA (below-left of MC dropout)
% ============================================================
\node [block, auaaColor, below=2.0cm of mcdropout, xshift=-4.0cm, minimum width=5.2cm, minimum height=1.7cm] (auaa) {
    \textbf{AUAA}\\[3pt]
    \scriptsize $\boldsymbol{\alpha}_t = \text{softmax}\!\left(\frac{\mathbf{Q}\mathbf{K}^\top}{\sqrt{d_k}}\right) \odot \frac{1}{1+\boldsymbol{\sigma}_t}$\\[3pt]
    {\scriptsize Uncertainty gates attention weights}
};

% ============================================================
% Dynamic Weight Adapter (below-right of MC dropout)
% ============================================================
\node [block, adapterColor, below=2.0cm of mcdropout, xshift=4.0cm, minimum width=5.2cm, minimum height=1.7cm] (adapter) {
    \textbf{Dynamic Weight Adapter}\\[3pt]
    \scriptsize $s_t = \alpha_t \cdot \mathbf{e}_t + (1-\alpha_t) \cdot \mathbf{u}_t$\\[3pt]
    {\scriptsize GRU-context learns $\alpha_t$ over time}
};

% ============================================================
% Feature Attribution (below MC dropout, centred)
% ============================================================
\node [block, attribColor, below=3.0cm of mcdropout, yshift=-1.8cm, minimum width=5.2cm, minimum height=1.7cm] (attribution) {
    \textbf{Feature Attribution}\\[3pt]
    \scriptsize $\mathbf{w}_f = \text{softmax}(\mathbf{V}\!\cdot\!\tanh(\mathbf{W}\mathbf{x}+\mathbf{b}))$\\[3pt]
    {\scriptsize Per-sensor importance + Chebyshev bound}
};

% ============================================================
% OUTPUT (below attribution)
% ============================================================
\node [block, outputColor, below=0.8cm of attribution, minimum width=8.0cm, minimum height=1.4cm] (output) {
    \textbf{Output}\\[3pt]
    {\scriptsize Anomaly Score $s_t$ \quad | \quad Per-sensor attribution $\mathbf{w}_f$}\\[3pt]
    {\scriptsize $P(\text{FP}) \leq 1/k^2$ \,(Chebyshev distribution-free FP guarantee)}
};
