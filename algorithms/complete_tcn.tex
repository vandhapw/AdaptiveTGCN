\begin{figure}[h!]
\centering
\resizebox{\linewidth}{!}{%
\begin{tikzpicture}[
    scale=0.9,
    layer/.style={rectangle, draw=black, thick, minimum width=10cm, minimum height=1.2cm},
    block/.style={rectangle, draw=black, thick, fill=tcncolor, minimum width=9cm, minimum height=2.5cm},
    conv/.style={rectangle, draw=black, fill=blue!20, minimum width=2cm, minimum height=0.6cm},
    node distance=0.8cm
]

% Input Layer
\node[layer, fill=inputcolor] (input) {
    \textbf{Input:} $\mathbf{X} \in \mathbb{R}^{B \times T \times C_{in}}$ \\
    Batch Size: $B$, Sequence Length: $T=window_size$, Input Channels: $C_{in}$
};

% Permutation
\node[layer, fill=gray!20, below=of input] (permute) {
    \textbf{Permute:} $(B, T, C_{in}) \rightarrow (B, C_{in}, T)$
};

% Temporal Block 1
\node[block, below=of permute] (tb1) {
    \textbf{Temporal Block 1} \vspace{0.3cm} \\
    Dilation : $d = 2^0 = 1$ \vspace{0.3cm} \\
    Channels \vspace{0.3cm}\\
    Dropout
};

% Temporal Block 2
\node[block, below=of tb1] (tb2) {
    \textbf{Temporal Block 2} \vspace{0.3cm} \\
    Dilation: $d = 2^1 = 2$ \vspace{0.3cm} \\
    Channels \vspace{0.3cm} \\
    Dropout
};

% Temporal Block 3
\node[block, below=of tb2] (tb3) {
    \textbf{Temporal Block 3} \vspace{0.3cm} \\
    Dilation: $d = 2^2 = 4$ \vspace{0.3cm} \\
    Channels \vspace{0.3cm} \\
    Dropout
};

% FC Layer
\node[layer, fill=orange!20, below=of tb3] (fc) {
    \textbf{Fully Connected:} $\mathbb{R}^{64} \rightarrow \mathbb{R}^{n}$ \\
    Extract last timestep: $\mathbf{x}[:, :, -1]$
};

% Output
\node[layer, fill=anomalycolor, below=of fc] (output) {
    \textbf{Output:} $\mathbf{y} \in \mathbb{R}^{B \times n}$ \\
    Predictions for: Output Features
};

% Connections
\draw[->, thick] (input) -- (permute);
\draw[->, thick] (permute) -- (tb1);
\draw[->, thick] (tb1) -- (tb2);
\draw[->, thick] (tb2) -- (tb3);
\draw[->, thick] (tb3) -- (fc);
\draw[->, thick] (fc) -- (output);

\end{tikzpicture}
}

\caption{Temporal Block Component of TCN}
