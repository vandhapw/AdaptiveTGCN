\begin{algorithm}[H]
\footnotesize
\caption{Model Training Procedure}
\label{alg:model_training}
\begin{algorithmic}
\Require Training set $\mathcal{D}_{train}$, validation set $\mathcal{D}_{val}$, learning rate $\eta$, epochs $E$, batch size $B$, patience $P$
\Ensure Optimal parameters $\boldsymbol{\theta}^*$

\State Initialize $\boldsymbol{\theta}$ (He init), optimizer $\gets$ Adam($\eta$)
\State $L_{best} \gets \infty$, $patience \gets 0$

\For{epoch $e = 1$ to $E$}
    \State Shuffle $\mathcal{D}_{train}$
    \For{each mini-batch $(\mathbf{X}_b, \mathbf{y}_b)$ of size $B$}
        \State $\hat{\mathbf{y}}_b \gets f_{\boldsymbol{\theta}}(\mathbf{X}_b)$
        \State $L \gets \mathcal{L}_{\text{MSE}}(\mathbf{y}_b, \hat{\mathbf{y}}_b) + \lambda_1 \mathcal{L}_{\text{unc}} + \lambda_2 \mathcal{L}_{\text{att}}$
        \State $\boldsymbol{\theta} \gets \boldsymbol{\theta} - \eta \cdot \nabla_{\boldsymbol{\theta}} L$
    \EndFor
    
    \State $L_{val} \gets \mathcal{L}_{\text{MSE}}(f_{\boldsymbol{\theta}}(\mathcal{D}_{val}), \mathbf{y}_{val})$
    \If{$L_{val} < L_{best}$}
        \State $L_{best} \gets L_{val}$, save $\boldsymbol{\theta}^* \gets \boldsymbol{\theta}$, $patience \gets 0$
    \Else
        \State $patience \gets patience + 1$
    \EndIf
    \If{$patience \ge P$} \textbf{break} \EndIf
\EndFor

\State \textbf{Return} $\boldsymbol{\theta}^*$
\end{algorithmic}
\end{algorithm}
