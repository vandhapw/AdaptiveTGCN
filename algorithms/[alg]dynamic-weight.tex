\begin{algorithm}[H]
\footnotesize
\caption{Dynamic Weight Adaptation for Hybrid Anomaly Scoring}
\label{alg:dynamic_weight}
\begin{algorithmic}
\Require Reconstruction error $\mathbf{e} \in \mathbb{R}^{B \times T}$, uncertainty $\mathbf{u} \in \mathbb{R}^{B \times T}$
\Require Encoders $\mathbf{W}_e, \mathbf{W}_u$, GRU, weight adapter $\mathbf{W}_\alpha, b_\alpha$
\Ensure Dynamic weight $\alpha_t \in [0,1]$, hybrid score $s_t$

\State \textbf{Step 1: Encode error and uncertainty independently}
\State $\mathbf{z}_e \gets \text{ReLU}(\mathbf{W}_e \cdot \mathbf{e} + b_e)$
\State $\mathbf{z}_u \gets \text{ReLU}(\mathbf{W}_u \cdot \mathbf{u} + b_u)$

\State \textbf{Step 2: Temporal context via GRU}
\State $\mathbf{c}_t \gets \text{Concat}(\mathbf{z}_{e,t},\, \mathbf{z}_{u,t})$
\For{$t = 1$ to $T$}
    \State $\mathbf{h}_t \gets \text{GRU}(\mathbf{c}_t,\, \mathbf{h}_{t-1})$
\EndFor

\State \textbf{Step 3: Predict weight and compute score}
\For{$t = 1$ to $T$}
    \State $\mathbf{f}_t \gets \text{Concat}(\mathbf{e}_t,\, \mathbf{u}_t,\, \mathbf{h}_t)$
    \State $\alpha_t \gets \text{sigmoid}(\mathbf{W}_\alpha \cdot \mathbf{f}_t + b_\alpha)$
    \State $s_t \gets \alpha_t \cdot \mathbf{e}_t + (1 - \alpha_t) \cdot \mathbf{u}_t$
\EndFor

\State \textbf{Return} $(\boldsymbol{\alpha},\, \mathbf{s})$
\end{algorithmic}
\end{algorithm}
