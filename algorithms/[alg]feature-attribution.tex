\begin{algorithm}[H]
\footnotesize
\caption{Feature-Level Anomaly Attribution}
\label{alg:feature_attribution}
\begin{algorithmic}
\Require Input $\mathbf{X} \in \mathbb{R}^{B \times T \times n}$, reconstruction errors $\mathbf{E} \in \mathbb{R}^{B \times T \times n}$
\Require Learnable params $\mathbf{W}, \mathbf{V}, b$, correlation matrix $\mathbf{C} \in \mathbb{R}^{n \times n}$
\Ensure Feature importance $\mathbf{w} \in \mathbb{R}^{n}$, weighted errors $\mathbf{E}_{w}$, anomaly flags $\mathbf{A}$

\State \textbf{Step 1: Inter-sensor correlation}
\State $\mathbf{C}_{ij} \gets \text{Corr}(\mathbf{X}_{:,:,i},\, \mathbf{X}_{:,:,j})$ \Comment{Physical sensor coupling}
\State $\mathbf{X}_{corr} \gets \mathbf{X} \times \mathbf{C}$

\State \textbf{Step 2: Temporal aggregation}
\State $\mathbf{X}_{agg} \gets \text{Mean}(\mathbf{X}_{corr},\ \text{dim}=T)$ \Comment{$B \times n$}

\State \textbf{Step 3: Learnable importance}
\For{$b = 1$ to $B$}
    \State $\mathbf{h}_b \gets \tanh(\mathbf{W} \cdot \mathbf{X}_{agg,b} + b)$
    \State $\mathbf{w}_b \gets \text{softmax}(\mathbf{V} \cdot \mathbf{h}_b)$ \Comment{$\sum_i w_{b,i} = 1$}
\EndFor
\State $\mathbf{w} \gets \frac{1}{B}\sum_{b=1}^{B} \mathbf{w}_b$ \Comment{Cross-batch average}

\State \textbf{Step 4: Weighted errors \& anomaly flags}
\For{$f = 1$ to $n$}
    \State $\mathbf{E}_{w}[:,:,f] \gets \mathbf{w}[f] \cdot \mathbf{E}[:,:,f]$
    \State $\mathbf{A}_f \gets \mathbb{1}\big[\mathbf{E}_{w}[:,:,f] > \tau_f\big]$
\EndFor

\State \textbf{Step 5: Attribution summary}
\State $\text{rank} \gets \text{argsort}(\mathbf{w},\ \text{descending})$
\State $\text{contrib}_i \gets \frac{\mathbf{w}_i \cdot \sum_{t,b} \mathbf{E}[b,t,i]}{\sum_{f,t,b} \mathbf{E}[b,t,f]} \times 100\%$

\State \textbf{Return} $(\mathbf{w},\, \mathbf{E}_{w},\, \mathbf{A})$
\end{algorithmic}
\end{algorithm}
